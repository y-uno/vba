Sub CombineCellsWithSpace()
    Dim ws As Worksheet
    Dim lastRow As Long
    Dim i As Long
    
    ' 対象のワークシートを指定（適切なワークシート名に変更してください）
    Set ws = ThisWorkbook.Sheets("Sheet1")
    
    ' 最終行を取得
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    
    ' 各行ごとに処理を実行
    For i = 1 To lastRow
        ws.Cells(i, 1).Value = ws.Cells(i, 1).Value & " [" & ws.Cells(i, 6).Value & "] [" & ws.Cells(i, 7).Value & "]"
    Next i
End Sub

Sub ClearValues()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Sheet1") ' シート名を適切なものに変更

    ' A1:G10 の 70 セルをクリア
    ws.Range("A1:G10").ClearContents
End Sub

Sub CopyValuesToSheet2()
    Dim ws1 As Worksheet
    Dim ws2 As Worksheet
    Dim x As Long ' 行番号
    Dim i As Long ' ループ変数
    
    ' シート名を適切なものに変更
    Set ws1 = ThisWorkbook.Sheets("Sheet1")
    Set ws2 = ThisWorkbook.Sheets("Sheet2")
    
    ' x に行番号を設定（適切な行番号を指定）
    x = 5
    
    ' i に行番号を設定（適切な行番号を指定）
    i = 1
    
    ' (x,5) の値を (i,8) にコピー
    ws2.Cells(i, 8).Value = ws1.Cells(x, 5).Value
    
    ' (x,8) の値を (i,9) にコピー
    ws2.Cells(i, 9).Value = ws1.Cells(x, 8).Value
End Sub
